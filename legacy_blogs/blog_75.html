  3 trizip :: [a] -> [b] -> [c] -> [(a,b,c)]
  4 trizip a b c
  5         | null a = []
  6         | null b = []
  7         | null c = []
  8 trizip (x:xs) (y:ys) (z:zs) = (++) [(x,y,z)] (trizip xs ys zs)
