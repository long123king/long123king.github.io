 169: void determineNextPos()
 170: {
 171:     if (endPoint._y == y >> abs(endPoint._x - x) < abs(speed))
 172:     {
 173:         startPoint = endPoint;

 175:

 177:         calcSpeed();
 178:  
 179:         x = startPoint._x;
 180:         y = startPoint._y;

 183:     {
 184:         startPoint = endPoint;
