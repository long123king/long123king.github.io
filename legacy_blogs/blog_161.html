   5:     {

   8:     for (; _ISORT_MAX < (_Count = _Last - _First) && 0 < _Ideal; )

  14:             _Unguarded_partition(_First, _Last, _Pred);

  18:

  20:             _Sort(_First, _Mid.first, _Ideal, _Pred), _First = _Mid.second;
  21:         else

  23:     }
  24:

  30:     }
  31:     else if (1 < _Count)

  33:     }
