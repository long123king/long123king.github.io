void dumpfile(const char* src, const char* dst)
{
    FILE* proc_status = fopen(src, "r");
    FILE* dst_dump = fopen(dst, "w+");
    if (dst_dump != NULL
        &&proc_status != NULL)
    {
        char ch = fgetc(proc_status);    
        while (ch != EOF )
        {
            fputc(ch, dst_dump);    
            ch = fgetc(proc_status);
        }            
        fclose(dst_dump);
        fclose(proc_status);
    } 
}
 
void my_terminate_handler()
{
    printf("my terminate handler is invoked!\n");
    pid_t pid = getpid();
    char path[512] = {0,};
    sprintf(path, "/proc/%d/status", pid);
    dumpfile(path, "proc_status");
    sprintf(path, "/proc/%d/task", pid);
 
    DIR           *d;
     struct dirent *dir;
    d = opendir(path);
    if (d)
    {
        while ((dir = readdir(d)) != NULL)
        {
            if (dir->d_type == DT_DIR)
            {
            printf("%s\n", dir->d_name);
            sprintf(path, "/proc/%d/task/%s/stack", pid, dir->d_name);
            char dst_path[256] = {0,};
            sprintf(dst_path, "proc_task_%s_stack", dir->d_name);
            dumpfile(path, dst_path);
            }
        }
        closedir(d);
    }
 
    //dumpfile(path, "proc_task");
    
    abort();
}
